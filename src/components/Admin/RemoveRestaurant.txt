import React, { useState, useEffect } from 'react';

function RemoveRestaurant() {
  const [restaurantName, setRestaurantName] = useState('');
  const [restaurants, setRestaurants] = useState([]);

  useEffect(() => {
    // Fetch restaurants from localStorage
    const storedRestaurants = JSON.parse(localStorage.getItem('restaurants')) || [];
    setRestaurants(storedRestaurants);
  }, []);

  const handleSubmit = (e) => {
    e.preventDefault();

    const updatedRestaurants = restaurants.filter(
      (restaurant) => restaurant.name !== restaurantName
    );

    if (updatedRestaurants.length === restaurants.length) {
      alert(`Restaurant "${restaurantName}" not found.`);
    } else {
      // Save the updated list to localStorage
      localStorage.setItem('restaurants', JSON.stringify(updatedRestaurants));
      alert(`Restaurant "${restaurantName}" removed successfully!`);
      setRestaurantName('');
      setRestaurants(updatedRestaurants); // Update the table
    }
  };

  return (
    <div className="admin-container">
      <h1>Remove Restaurant</h1>
      <form onSubmit={handleSubmit}>
        <div className="form-group">
          <label>
            Select Restaurant to Remove:
            <select
              value={restaurantName}
              onChange={(e) => setRestaurantName(e.target.value)}
              required
            >
              <option value="">-- Select Restaurant --</option>
              {restaurants.map((restaurant, index) => (
                <option key={index} value={restaurant.name}>
                  {restaurant.name}
                </option>
              ))}
            </select>
          </label>
        </div>

        <button type="submit">Remove Restaurant</button>
      </form>
    </div>
  );
}

export default RemoveRestaurant;
