import React, { useState } from 'react';

function TableAvailability() {
  const [tables, setTables] = useState([
    { tableNumber: 1, isAvailable: true },
    { tableNumber: 2, isAvailable: false },
    { tableNumber: 3, isAvailable: true },
    { tableNumber: 4, isAvailable: false },
  ]);

  const toggleAvailability = (index) => {
    const updatedTables = [...tables];
    updatedTables[index].isAvailable = !updatedTables[index].isAvailable;
    setTables(updatedTables);
    alert(
      `Table ${updatedTables[index].tableNumber} is now ${
        updatedTables[index].isAvailable ? 'Available' : 'Occupied'
      }.`
    );
  };

  return (
    <div className="restaurant-admin-container">
      <h1>Table Availability</h1>
      <ul>
        {tables.map((table, index) => (
          <li key={index}>
            Table {table.tableNumber} -{' '}
            <strong>{table.isAvailable ? 'Available' : 'Occupied'}</strong>
            <button onClick={() => toggleAvailability(index)}>
              Mark as {table.isAvailable ? 'Occupied' : 'Available'}
            </button>
          </li>
        ))}
      </ul>
    </div>
  );
}

export default TableAvailability;
